
\begin{itemize}
    \item \textbf{Arithmetic Progression (AP):}
    Sum of the first $n$ terms:
    \[ \sum_{i=1}^{n} i = 1 + 2 + \dots + n = \frac{n(n+1)}{2} \]

    \item \textbf{Sum of Squares:}
    Sum of the first n squares:
    \[ \sum_{i=1}^{n} i^2 = 1^2 + 2^2 + \dots + n^2 = \frac{n(n+1)(2n+1)}{6} \]

    \item \textbf{Sum of Cubes:}
    Sum of the first n cubes:
    \[ \sum_{i=1}^{n} i^3 = 1^3 + 2^3 + \dots + n^3 = \left( \frac{n(n+1)}{2} \right)^2 \]

    \item \textbf{Geometric Progression (GP):}
    Sum of the first $n$ terms (ratio $r \neq 1$):
    \[ \sum_{i=0}^{n-1} a \cdot r^i = a \left( \frac{r^n - 1}{r - 1} \right) \]
    \textit{Note:} If $|r| < 1$, the infinite sum is $S_{\infty} = \frac{a}{1-r}$.

    \item \textbf{Arithmetico-Geometric Series:}
    Sum of $\sum_{i=1}^{n} i \cdot r^{i-1}$ (common in expected value problems):
    \[ \sum_{i=1}^{n} i \cdot r^{i-1} = \frac{1 - r^n}{(1-r)^2} - \frac{n r^n}{1-r} \]

\end{itemize}
