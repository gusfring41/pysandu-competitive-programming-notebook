\subsection{Circles}

\begin{itemize}
    \item \textbf{Area: } $\pi r^2$
    \item \textbf{Circumference: } $2 \pi r$
    \item \textbf{Equation: } $(x-h)^2 + (y-k)^2 = r^2$
\end{itemize}

\subsection{Spheres}

\begin{itemize}
    \item \textbf{Volume: } $\frac{4}{3} \pi r^3$
    \item \textbf{Surface Area: } $4 \pi r^2$
    \item \textbf{Equation: } $(x-h)^2 + (y-k)^2 + (z-l)^2 = r^2$
\end{itemize}
